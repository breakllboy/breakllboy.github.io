using namespace std;
int ans = 0;
int n;
bool a[20];
bool x1[20];
bool y1[20];
void dfs(int x)
{
if(x>n)
{
ans++;
return;
}
for(int i = 1;i <= n;i++)
{
if(x1[i+x] == false&&y1[i-x+n] == false&&a[i] == false)
{
x1[i+x] = true;
y1[i-x+n] = true;
a[i] = true;
dfs(x+1);
x1[i+x] = false;
y1[i-x+n] = false;
a[i] = false;
}
}
}
int main()
{
cin>>n;
memset(a,false,sizeof(a));
memset(x1,false,sizeof(x1));
memset(y1,false,sizeof(y1));
dfs(1);
cout<<ans<<endl;
return 0;
}
